54// Does the string contain slash?
55constexpr bool containss_slash(const char* s) {
56 return *s ? (*s == '/' ? true : containss_slash(s + 1)) : false;
57}
58// Returns the next position after the right-most slash
59constexpr const char* after_right_slash(const char* s) {
60 return (*s == '/') ? (s + 1) : after_right_slash(s - 1);
61}
62// Gets the tail of string
63constexpr const char* tail(const char* s) {
64 return *s ? tail(s + 1) : s;
65}
67
73constexpr const char* pathToFilename(const char* path) {
74 return (path && path[0]) ? (containss_slash(path) ? after_right_slash(tail(path)) : path) : "";
75}
