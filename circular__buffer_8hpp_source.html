233 size_t read(value_type* outbuf, const size_t num) {
234 size_t sz = std::min(num, size());
235 if (sz == 0) {
236 return sz;
237 }
238 auto tail = _tail;
239 auto src = &_buf[tail];
240 size_t elms = std::min(_cap - tail, sz);
241
242 std::copy(src, src + elms, outbuf);
243 tail = (tail + elms) % _cap;
244 size_t ret = elms;
245
246 if (elms < sz) {
247 outbuf += elms;
248 src = &_buf[tail];
249 elms = sz - elms;
250
251 std::copy(src, src + elms, outbuf);
252 ret += elms;
253 }
254 return ret;
255 }
