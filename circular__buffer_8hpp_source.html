255 size_t read(value_type* outbuf, const size_t num) {
256 size_t sz = std::min(num, size());
257 if (sz == 0) {
258 return sz;
259 }
260 auto tail = _tail;
261 auto src = &_buf[tail];
262 size_t elms = std::min(_cap - tail, sz);
263
264 std::copy(src, src + elms, outbuf);
265 tail = (tail + elms) % _cap;
266 size_t ret = elms;
267
268 if (elms < sz) {
269 outbuf += elms;
270 src = &_buf[tail];
271 elms = sz - elms;
272
273 std::copy(src, src + elms, outbuf);
274 ret += elms;
275 }
276 return ret;
277 }
